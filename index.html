🚀 原生 JS vs jQuery DOM 操作对比

JS 原生 JavaScript DOM 操作

1. 元素选择

// 通过 ID 选择
document.getElementById('id');
document.querySelector('#id');

// 通过 class 选择
document.getElementsByClassName('class');
document.querySelectorAll('.class');

// 通过标签选择
document.getElementsByTagName('div');
document.querySelectorAll('div');
document.querySelector('div'); //返回第一个

2. 修改内容

// 修改文本内容
element.textContent = '新文本';
element.innerText = '新文本';

// 修改 HTML
element.innerHTML = '<strong>HTML</strong>';

// 修改表单值
input.value = '新值';

3. 修改样式

// 直接修改 style
element.style.color = 'red';
element.style.backgroundColor = 'blue';

// 添加/删除 class
element.classList.add('active');
element.classList.remove('active');
element.classList.toggle('active');

4. 创建和插入元素

// 创建元素
const div = document.createElement('div');
div.textContent = '新元素';

// 插入元素
parent.appendChild(div);
parent.insertBefore(newNode, referenceNode);
parent.append(div);
parent.prepend(div);

5. 事件绑定

// 添加事件监听
element.addEventListener('click', function(e) { console.log('点击了!'); });

// 移除事件监听
element.removeEventListener('click', handler);

// 事件委托
parent.addEventListener('click', function(e) { if (e.target.matches('.item')) { // 处理点击 } });

6. 属性和数据操作

// 获取/设置属性
element.getAttribute('id');
element.setAttribute('data-id', '123');
element.removeAttribute('disabled');

// 自定义数据属性
element.dataset.userId = '456';
const id = element.dataset.userId;

jQuery jQuery DOM 操作

1. 元素选择

// 通过 ID 选择
$('#id');

// 通过 class 选择
$('.class');

// 通过标签选择
$('div');

// 复杂选择器
$('ul li:first');
$('input[type="text"]');
$('.class1, .class2');

2. 修改内容

// 修改文本内容
$('#element').text('新文本');

// 修改 HTML
$('#element').html('<strong>HTML</strong>');

// 修改表单值
$('input').val('新值');

// 获取内容
const text = $('#element').text();

3. 修改样式

// 修改 CSS
$('#element').css('color', 'red');
$('#element').css({ color: 'red', backgroundColor: 'blue' });

// 操作 class
$('#element').addClass('active');
$('#element').removeClass('active');
$('#element').toggleClass('active');
$('#element').hasClass('active');

4. 创建和插入元素

// 创建元素
const $div = $('<div>新元素</div>');

// 插入元素
$('#parent').append($div);
$('#parent').prepend($div);
$div.appendTo('#parent');
$div.prependTo('#parent');

// 在特定位置插入
$('#ref').before($div);
$('#ref').after($div);

5. 事件绑定

// 绑定点击事件
$('#btn').on('click', function() { console.log('点击了!'); });

// 简写形式
$('#btn').click(function() { ... });

// 事件委托
$('#parent').on('click', '.item', function() { // 处理点击 });

// 解除绑定
$('#btn').off('click');

6. 属性和数据操作

// 获取/设置属性
$('#element').attr('id');
$('#element').attr('data-id', '123');
$('#element').removeAttr('disabled');

// 自定义数据属性
$('#element').data('userId', '456');
const id = $('#element').data('userId');
